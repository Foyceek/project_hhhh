entity sound_synth is
    Port ( melody : in STD_LOGIC_VECTOR (2 downto 0);
           cnt : in STD_LOGIC_VECTOR (5 downto 0);
	   rst : out STD_LOGIC);
           relay : out STD_LOGIC);
end sound_synth;

architecture Behavioral of sound_synth is

begin
    process (cnt)
    begin
	rst <= '0'       
          if melody=x"0" then
            if cnt = x"0" then
                relay <= '1';
            elsif cnt = x"1" then
                relay <= '0';
            elsif cnt = x"2" then
                relay <= '1';
            elsif cnt = x"3" then
                relay <= '0';
            elsif cnt = x"4" then
                relay <= '1';

                relay <= '1';

                relay <= '1';
            elsif cnt = x"13" then
                relay <= '0';

                relay <= '1';
            else 
                relay <= '0';
		rst <= '1'
            end if;
            
          elsif melody=x"1" then
            if cnt = x"0" then
                relay <= '1';
            elsif cnt = x"1" then
                relay <= '0';
            elsif cnt = x"2" then
                relay <= '1';
            elsif cnt = x"3" then

                relay <= '0';
            elsif cnt = x"6" then
                relay <= '1';
            elsif cnt = x"7" then
                relay <= '0';
            elsif cnt = x"8" then
                relay <= '1';
            elsif cnt = x"9" then
                relay <= '0';
            elsif cnt = x"10" then
                relay <= '1';
            elsif cnt = x"11" then
                relay <= '0';
            elsif cnt = x"12" then
                relay <= '1';
            elsif cnt = x"13" then
                relay <= '0';
            elsif cnt = x"14" then
                relay <= '1';
            else 
                relay <= '0';
		rst <= '1'
            end if;
            
          elsif melody=x"2" then
            if cnt = x"0" then
                relay <= '1';
            elsif cnt = x"1" then
                relay <= '0';
            elsif cnt = x"2" then
                relay <= '1';
            elsif cnt = x"3" then
                relay <= '0';
            elsif cnt = x"4" then
                relay <= '1';
            elsif cnt = x"5" then
                relay <= '0';
            elsif cnt = x"6" then
                relay <= '1';
            elsif cnt = x"7" then
                relay <= '0';
            elsif cnt = x"8" then
                relay <= '1';
            elsif cnt = x"9" then
                relay <= '0';
            elsif cnt = x"10" then
                relay <= '1';
            elsif cnt = x"11" then
                relay <= '0';
            elsif cnt = x"12" then
                relay <= '1';
            elsif cnt = x"13" then

            end if;

            elsif cnt = x"1" then

                relay <= '1';
            elsif cnt = x"9" then
                relay <= '0';
            elsif cnt = x"10" then
                relay <= '1';
            elsif cnt = x"11" then
                relay <= '0';
            elsif cnt = x"12" then
                relay <= '1';
            elsif cnt = x"13" then
                relay <= '0';
            elsif cnt = x"14" then
                relay <= '1';
